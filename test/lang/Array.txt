public class Array<T> : List<T> {

    private final int length;
    public int Length {
        get {
            return length;
        }
    }

    public T this[int i] {
        get {
            native (return) {
                return data[v_i];
            }
        }
        set {
            native (source) {
                data[v_i] = v_value;
            }
        }
    }

    native (header) {
        g_T* data;
        create(g_T* list, long size);
    }

    native (source) {
        template <typename g_T>
        lang::Array<g_T>::create(g_T* list, long size) {
            this->data = static_cast<g_T*>(system::memory::array(sizeof(g_T), size));

            if (this->data == nullptr) {
                this->f_length = 0;
            } else {
                this->f_length = size;
                for (long i = 0; i < size; i++) {
                    this->data[i] = list[i];
                }
            }
        }
    }

    public this() {
        this.length = 0;

        native (source) {
            this->data = nullptr;
        }
    }

    public this(int length) {
        native (source) {
            this->data = static_cast<g_T*>(system::memory::array(sizeof(g_T), v_length));

            if (this->data == nullptr) {
                this->f_length = 0;
            } else {
                this->f_length = v_length;
                for (long i = 0; i < v_length; i++) {
                    this->data[i] = generic<g_T>::def();
                }
            }
        }
    }
}
