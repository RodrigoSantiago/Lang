namespace lang;

public sync class Object {
    native (header) {
        lang::weakptr* weak;

        void * operator new(unsigned int size) { return lang::memory::alloc(size); }
        void operator delete(void * p) { return lang::memory::dealloc(p); }
    }

    public this() {
        native (source) {
            weak = lang::memory::weak(reinterpret_cast<void*>(this));
        }
    }

    public ~this() {
    }

    public string toString() {
        return "Object[" + hash() + "]";
    }

    public int hash() {
        return 0;
    }

    public bool equals(let Object other) {
        return this == other;
    }
}
