public class Tuple<T> : List<T> {
    private T[] values;

    public int Length {
        get {
            return values.Length;
        }
    }

    public T this[int i] {
        let {
            return values[i];
        }
    }

    public this(arg T[] values) {
        this.values = new Array<T>();
    }

    public this(arg T@[] values) {
        this.values = new Array<T>();
    }
}
